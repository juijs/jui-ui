setOption와 reload 함수를 사용하여 select 이벤트를 설정하면, 아래와 같이 날짜의 선택에 따라 다른 datepicker에 영향을 줄 수 있습니다.

SMTWTFS 

event: {
    select: function(date, e) {
        // 생략
        datepicker7.setOption('maxDate', new Date(date));
        datepicker7.reload();
    }
}
