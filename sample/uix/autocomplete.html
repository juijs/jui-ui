Auto Complete

텍스트박스에 문자열을 입력했을 경우, 해당 값이 포함된 키워드 목록을 보여주는 UI 컴포넌트입니다.

Default

아래 키워드 중에 입력한 문자가 포함되어 있으면, 드롭다운 메뉴에 키워드 목록이 나타납니다.  

var ac_words = [
  "ActionScript", "AppleScript", "Asp", "BASIC", "C", "C++",
  "Clojure", "COBOL", "ColdFusion", "Erlang", "Fortran", "Groovy",
  "Haskell", "Java", "JavaScript", "Lisp", "Perl", "PHP",
  "Python", "Ruby", "Scala", "Scheme"
];

jui.ready(function(ui, uix, _) {
	ac_1 = uix.autocomplete(".ac", {
		target: "input[type=text]",
		words: ac_words,
		event: {
			change: function(text) {
				alert(text);
			}
		}
	});
});

아래는 자동완성 컴포넌트의 마크업과 템플릿입니다. 셀렉터는 그룹 태그이며 텍스트박스는 target 옵션을 통해 정할 수 있습니다.

<span class="ac group">
	<label class="label label-small"><i class="icon-search"></i></label>
	<input type="text" class="input input-small" style="width: 100px" />
</span>

<script data-jui=".ac" data-tpl="words" type="text/template">
	<div class="dropdown">
		<ul>
		<! for(var i = 0; i < words.length; i++) { !>
			<li><!= words[i] !></li>
		<! } !>
		</ul>
	</div>
</script>

Update

Run - 좌측 버튼을 클릭하면 아래와 같은 단어 목록으로 변경합니다.  

var ac_words_hangul = [
	"자바의 정석",
	"열혈강의 C++",
	"유쾌한 심리학"
];

ac_1[1].update(ac_words_hangul);
